function Cover({ url, title }) {
  if (url) {
    return (
      <img
        src={url}
        alt={title}
        loading="lazy"
        decoding="async"
        style={{ width: '100%', aspectRatio: '3/4.3', objectFit: 'cover', borderRadius: T.radius.lg }}
      />
    );
  }
  return (
    <div
      style={{
        width: '100%',
        aspectRatio: '3/4.3',
        borderRadius: T.radius.lg,
        background: coverGradient(title),
        display: 'flex',
        alignItems: 'flex-end',
        padding: 16,
        color: '#dff5ef',
        fontSize: 15,
        fontWeight: 700,
        lineHeight: 1.3,
        boxSizing: 'border-box',
      }}
    >
      {title}
    </div>
  );
}
